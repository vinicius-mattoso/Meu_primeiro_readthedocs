def generate_documentation(module_name, output_dir):
    '''
    Generates HTML documentation for the given Python module and saves it in the specified directory.

    Parameters:
        module_name (str): The name of the Python module/package.
        output_dir (str): The directory where the documentation will be saved.
    '''
    # Prepare the command and execute it
    command = ['python', '-m', 'pydoc', '-w', module_name]
    
    # Change directory to the parent of the module directory
    parent_dir = os.path.dirname(output_dir)
    result = subprocess.run(command, stdout=subprocess.PIPE, stderr=subprocess.PIPE, cwd=parent_dir)

    if result.returncode != 0:
        print(f"Failed to generate documentation for {module_name}")
    else:
        print(f"Documentation generated for {module_name}")

def move_html_files(src_dir, dest_dir):
    '''
    Move HTML files starting with 'biganchoring_ml_module' from the source directory
    to the destination directory.

    Args:
        src_dir (str): The source directory where the HTML files are located.
        dest_dir (str): The destination directory where the HTML files should be moved.
    '''
    # Ensure the destination directory exists
    if not os.path.exists(dest_dir):
        os.makedirs(dest_dir)

    # List all files in the source directory
    for file_name in os.listdir(src_dir):
        # Check if the file is an HTML file starting with 'biganchoring_ml_module'
        if file_name.startswith('biganchoring_ml_module') and file_name.endswith('.html'):
            # Define full file paths
            src_file = os.path.join(src_dir, file_name)
            dest_file = os.path.join(dest_dir, file_name)
            
            # Move the file
            shutil.move(src_file, dest_file)

def create_documentation():
    '''
    Creates documentation for all Python files in the current directory and the package itself,
    saving it in the documentation folder.

    Returns:
        None
    '''
    # Define the root directory of the module
    module_dir = os.path.dirname(os.path.abspath(__file__))
    documentation_dir = os.path.join(module_dir, 'documentation')

    # Create the documentation directory if it doesn't exist
    if not os.path.exists(documentation_dir):
        os.makedirs(documentation_dir)

    # Generate documentation for the package itself
    generate_documentation('biganchoring_ml_module', documentation_dir)

    # Iterate over all files in the module directory
    for root, _, files in os.walk(module_dir):
        for file in files:
            if file.endswith('.py') and file != '__init__.py':
                module_path = os.path.join(root, file)
                module_name = os.path.relpath(module_path, module_dir).replace(os.path.sep, '.').replace('.py', '')
                if module_name.startswith('.'):
                    module_name = module_name[1:]
                if 'biganchoring_ml_module.' in module_name:
                    # Generate documentation for the module
                    generate_documentation(module_name, documentation_dir)
    
    # Move the generated HTML files to the documentation folder
    move_html_files(module_dir, documentation_dir)



if __name__ == "__main__":
    create_documentation()
